תְּאַרְיְנוּ: שְׁנֵים עָשָׂר דַּקּוֹת שֶׁל שַׁקֵּט מֻחְלָט — הַמִּשְׁפָּחָה מְרַחֶפֶת בְּנָחַת מֵעַל לִיעַר אֹרֶן צָפוּף, הָאֲוִיר קַרִיר וְרֵיחָנִי, בּוּדַפֶּשְׁט נִגְלֵית מִלְּמַטָּה בִּמְחִי-עַיִן. הַלִּיבֶגֶה (Libegő) — כִּסֵּא הָרוּחַ שֶׁל זוּגְלִיגֶט — נִפְתַּח בְּ-1970 וְהוּא אֶחָד הַחֲוָיוֹת הַמְּיֻחָדוֹת בְּבּוּדַפֶּשְׁט: מַעֲבִיר נוֹסְעִים 1.04 קִ"מ לְמַעְלָה עַד קִרְבַת פִּסְגַּת יָנוֹשׁ-הֶגִּי. הַמָּסְלוּל שֶׁל הַיּוֹם: עֲלִיָּה בַּלִּיבֶגֶה, טִיּוּל קַל לְנוֹרְמָפָה, אַחַר כָּךְ רַכֶּבֶת הַיְּלָדִים בַּחֲזָרָה — צִירוּף מֻשְׁלָם לְיֶלֶד מִכָּל גִּיל.

NORMAFA (ANNA ⭐ 4.8/5 (2,100+ דֵּעוֹת)

Normafa (Anna-rét) / נוֹרְמָפָה — מִגְרַשׁ אַנָּה-רֵת

נוֹרְמָפָה — מִישׁוֹר יַעַר בְּגֹבַה 477 מ' עַל פְּנֵי הַיָּם — הוּא מַטָּרָה שֶׁל תּוֹשָׁבֵי בּוּדַפֶּשְׁט בְּקַיִּץ וּבְחֹרֶף כְּאֶחָד. מִגְרַשׁ אַנָּה-רֵת (Anna-rét) הוּא לֵב הַפַּארְק מִבְּחִינַת יְלָדִים: מִגְרַשׁ מוֹדֶרְנִי עִם מִגְלָשׁוֹת עֵץ, מִתְקָנֵי טִיפּוּס, נְדְנָדוֹת לְכָל גִּיל, וּמַגְלֶשֶׁת קַיִּץ אֲרוּכָּה (גְּרֵירַת עֲגָלוֹן = יָרוּד קַיִּץ). שׁוּלְחָנוֹת פִּיקְנִיק בְּכָל עֵבֶר, עֵצֵי צֵל, וּבְקָרְבַת מָקוֹם — קִיּוֹסְק עִם לַנְגּוֹשׁ וְרֵיטֵשׁ חָם. הַיֶּלֶד הַקָּטָן (אִיתַי, 4) יִמְצָא מֵתַחֲנַת "נְדְנָדַת תִּינּוֹקוֹת" הִיא אֶחַת הַנְּדְנָדוֹת הַמּוּקָפוֹת הַבְּטוּחוֹת, וּמִגְרַשׁ הָעֵץ מַתְאִים בְּדִיּוּק לַטַּף.

שָׁעוֹת: פָּתוּחַ 24/7 (כּוּלּוֹ חָפְשִׁי, חִינָּם)

כַּרְטִיס: חִינָּם

Gyermekvasút / רַכֶּבֶת הַיְּלָדִים

מָתַי לְהִשְׁתַּמֵּשׁ: אֵם מֶזֶג אֲוִיר גָּרוּעַ, הַלִּיבֶגֶה סָגוּר (תַּחְזוּקָה בְּיוֹם שֵׁנִי אַחֵר), אוֹ אִם הַיְּלָדִים עֲיֵפִים מֻקְדָּם.

חֲלוּפָה: עֲלוּ בְּרַכֶּבֶת הַשִּׁינַּיִּית (Fogaskerekű, קַו 60) מֵ-Széll Kálmán tér עַד Széchenyi-hegy (18 דַּק'). מִשָּׁם: בְּחֲרוּ בְּרַכֶּבֶת הַיְּלָדִים כִּרְכִּיבָה קַלָּה בְּלִי הֲלִיכָה, אוֹ פִּיקְנִיק בַּחוֹרֶשׁ הַסָּמוּךְ. לַחֲזָרָה לַמֶּרְכָּז — אוֹתוֹ קַו 60 בַּכִּוּוּן הָהֵפֶךְ. Városmajor Park בְּמָחוֹז XII — פַּארְק צַלְמָנִי עִם מִגְרַשׁ מִשְׂחָקִים גָּדוֹל, שְׁבִילֵי חַצָּץ, וּמִרְבַּד דֶּשֶׁא לְפִּיקְנִיק — כ-15 דַּק' בְּרֶגֶל מֵ-Széll Kálmán tér.
